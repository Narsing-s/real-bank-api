<script>

const MULE_URL="http://localhost:8081/assistance/whatsapp";

function openChat(){

document.getElementById("chat").style.display="block";

}

async function sendMessage(event){

if(event.key!="Enter") return;

let input=document.getElementById("userInput");

let msg=input.value;

if(msg=="") return;

let body=document.getElementById("chatBody");

body.innerHTML+=`<div class="message">👤 ${msg}</div>`;

input.value="";

try{

let response=await fetch(MULE_URL,{

method:"POST",

headers:{
"Content-Type":"application/json"
},

body:JSON.stringify({

mobileNumber:"9876543210",

message:msg

})

});

let data=await response.json();

body.innerHTML+=`<div class="message">🤖 ${JSON.stringify(data)}</div>`;

body.scrollTop=body.scrollHeight;

}catch(e){

body.innerHTML+=`<div class="message">Connection Failed</div>`;

}

}

</script>
